/**
 * Description
 * @param {Callback} callback che si attiva dopo che la connessione è stabilita, o se c'è un errore nella connessione
 * @returns {callback} la funzione callback
 */
function connectToDb(cb) {
  MongoClient.connect(uri) //connessione al database mongo uri
    // è una promise e da accesso al Client connesso al database
    .then((client) => {
      db = client.db(); // ritorna la connessione al database. L'interfaccia del DB
      //console.log(db);
      return cb();
    })
    .catch((err) => {
      //console.error(err);
      return cb(err);
    });
}

app.get("/users", (req, res) => {
  //! PAGINATION VIDEO
  const page = req.query.page || 0;
  const maxDoc = 3;
  //? end pagination
  let result = [];
  // accede alla collezione indicata come argomento
  dbConn
    .collection("users")
    // ritorna un cursor, ossia un oggetto che punta ad un set di documenti. Da accesso ai metodo toArray (inserisce i documenti in un array) e forEach (processa i singoli elementi)
    .find()
    // i documenti sono estrapoli in Batch quindi ciò ottimizza il lavoro della network (101 alla volta)
    .sort({ name: -1 })
    //? Pagination part for skip pages
    .skip(page * maxDoc) //skip il numero di documenti per mostrare gli altri
    .limit(maxDoc)
    //? end pagination
    // async
    .forEach((user) => {
      result.push(user);
    })
    .then(() => {
      res.status(200).json({ result });
    })
    .catch(() => {
      res.status(500).json({ err: "Error" });
    });
});
 
app.get("/users/:id", (req, res) => {
  let id = req.params.id;
 
  if (ObjectId.isValid(id)) {
    dbConn
      .collection("users")
      // con ObjectId: id deve avere una sintassi specifica. Quando non esiste questo id ritorna null se non gestito il non-trovato
      //? BSONError: input must be a 24 character hex string, 12 byte Uint8Array, or an integer
 
      .findOne({ _id: new ObjectId(id) })
      .then((result) => {
        res.status(200).json(result);
      })
      .catch((err) => {
        res.status(500).json({ err: err });
      });
  } else {
    res.status(500).json({ error: "Id format is not valid" });
  }
});
